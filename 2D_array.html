            public class MatrixEquality {
                public static void main(String[] args) {

                    System.out.print("Enter number of columns: ");

                    int[][] matrix1 = new int[rows][cols];
                    int[][] matrix2 = new int[rows][cols];
            
                    System.out.println("Enter first matrix elements:");
                    fillMatrix(scanner, matrix1);
                    
                    System.out.println("Enter second matrix elements:");
                    fillMatrix(scanner, matrix2);
            
                    if (areEqual(matrix1, matrix2)) {
                        System.out.println("Matrices are equal.");
                    } else {
                        System.out.println("Matrices are not equal.");
                    }
                }
            
                private static void fillMatrix(Scanner scanner, int[][] matrix) {
                    for (int i = 0; i < matrix.length; i++) {
                        for (int j = 0; j < matrix[i].length; j++) {
                            matrix[i][j] = scanner.nextInt();
                        }
                    }
                }
            
                private static boolean areEqual(int[][] matrix1, int[][] matrix2) {
                    for (int i = 0; i < matrix1.length; i++) {
                        for (int j = 0; j < matrix1[i].length; j++) {
                            if (matrix1[i][j] != matrix2[i][j]) {
                                return false;
                            }
                        }
                    }
                    return true;
                }
            }

      Output: (Assuming user input for both matrices: 1 2; 3 4) Matrices are equal.

    • 11. Program to Determine Whether a Given Matrix is an Identity Matrix 
      import java.util.Scanner;
            
            public class IdentityMatrix {
                public static void main(String[] args) {
                    Scanner scanner = new Scanner(System.in);
                    System.out.print("Enter number of rows: ");
                    int rows = scanner.nextInt();
                    System.out.print("Enter number of columns: ");
                    int cols = scanner.nextInt();
            
                    int[][] matrix = new int[rows][cols];
                    boolean isIdentity = true;
            
                    System.out.println("Enter matrix elements:");
                    fillMatrix(scanner, matrix);
            
                    if (rows != cols) {
                        isIdentity = false;
                    } else {
                        for (int i = 0; i < rows; i++) {
                            for (int j = 0; j < cols; j++) {
                                if (i == j && matrix[i][j] != 1) {
                                    isIdentity = false;
                                } else if (i != j && matrix[i][j] != 0) {
                                    isIdentity = false;
                                }
                            }
                        }
                    }
            
                    if (isIdentity) {
                        System.out.println("The matrix is an identity matrix.");
                    } else {
                        System.out.println("The matrix is not an identity matrix.");
                    }
                }
            
                private static void fillMatrix(Scanner scanner, int[][] matrix) {
                    for (int i = 0; i < matrix.length; i++) {
                        for (int j = 0; j < matrix[i].length; j++) {
                            matrix[i][j] = scanner.nextInt();
                        }
                    }
                }
            }

      Output: (Assuming user input for matrix: 1 0; 0 1) The matrix is an identity matrix.

    • 13. Program to Determine Whether a Given Matrix is a Sparse Matrix 
      import java.util.Scanner;
            
            public class SparseMatrix {
                public static void main(String[] args) {
                    Scanner scanner = new Scanner(System.in);
                    System.out.print("Enter number of rows: ");
                    int rows = scanner.nextInt();
                    System.out.print("Enter number of columns: ");
                    int cols = scanner.nextInt();
            
                    int[][] matrix = new int[rows][cols];
                    int nonZeroCount = 0;
            
                    System.out.println("Enter matrix elements:");
                    fillMatrix(scanner, matrix);
            
                    for (int[] row : matrix) {
                        for (int element : row) {
                            if (element != 0) {
                                nonZeroCount++;
                            }
                        }
                    }
            
                    if (nonZeroCount < (rows * cols) / 2) {
                        System.out.println("The matrix is a sparse matrix.");
                    } else {
                        System.out.println("The matrix is not a sparse matrix.");
                    }
                }
            
                private static void fillMatrix(Scanner scanner, int[][] matrix) {
                    for (int i = 0; i < matrix.length; i++) {
                        for (int j = 0; j < matrix[i].length; j++) {
                            matrix[i][j] = scanner.nextInt();
                        }
                    }
                }
            }

      Output: (Assuming user input for a 3x3 matrix: 0 0 0; 0 1 0; 0 0 0) The matrix is a sparse matrix.
