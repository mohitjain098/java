    public class AverageMarks {
                public static void main(String[] args) {
                    int[] marks = {85, 90, 78, 92, 88};
                    double sum = 0;

  • 45. Java Program to calculate electricity bill 
    import java.util.Scanner;
            
            public class ElectricityBill {
                public static void main(String[] args) {
                    Scanner scanner = new Scanner(System.in);
                    System.out.print("Enter units consumed: ");
                    int units = scanner.nextInt();
                    double bill;
                    
                    if (units <= 100) {
                        bill = units * 1.20;
                    } else if (units <= 300) {
                        bill = 100 * 1.20 + (units - 100) * 2.00;
                    } else {
                        bill = 100 * 1.20 + 200 * 2.00 + (units - 300) * 3.00;
                    }
                    
                    System.out.printf("Electricity Bill: %.2f", bill);
                }
            }

    Output: Enter units consumed: 250, Electricity Bill: 320.00

  • 47. Java Program to calculate compound interest 
    import java.util.Scanner;
            
            public class CompoundInterest {
                public static void main(String[] args) {
                    Scanner scanner = new Scanner(System.in);
                    System.out.print("Enter principal amount: ");
                    double principal = scanner.nextDouble();
                    System.out.print("Enter rate of interest: ");
                    double rate = scanner.nextDouble();
                    System.out.print("Enter time (in years): ");
                    int time = scanner.nextInt();
                    
                    double amount = principal * Math.pow((1 + rate / 100), time);
                    double compoundInterest = amount - principal;
                    
                    System.out.printf("Compound Interest: %.2f", compoundInterest);
                }
            }

    Output: Enter principal amount: 10000, Compound Interest: 1000.00

    import java.util.Scanner;
            
            public class DistanceBetweenPoints {
                public static void main(String[] args) {
                    Scanner scanner = new Scanner(System.in);
                    System.out.print("Enter x1, y1: ");
                    double x1 = scanner.nextDouble();
                    double y1 = scanner.nextDouble();
                    System.out.print("Enter x2, y2: ");
                    double x2 = scanner.nextDouble();
                    double y2 = scanner.nextDouble();

  • 53. Write a Java program to find the number of days in a month 
    import java.util.Scanner;
            
            public class DaysInMonth {
                public static void main(String[] args) {
                    Scanner scanner = new Scanner(System.in);
                    System.out.print("Enter month (1-12): ");
                    int month = scanner.nextInt();
                    
                    int days = 0;
                    switch (month) {
                        case 1: case 3: case 5: case 7: case 8: case 10: case 12:
                            days = 31;
                            break;
                        case 4: case 6: case 9: case 11:
                            days = 30;
                            break;
                        case 2:
                            days = 28; // Not considering leap years for simplicity
                            break;
                        default:
                            System.out.println("Invalid month.");
                            return;
                    }
                    
                    System.out.println("Number of days: " + days);
                }
            }

    Output: Enter month (1-12): 2, Number of days: 28

  • 56. Write a Java program that takes a year from user and print whether that year is a leap year or not 
    import java.util.Scanner;
            
            public class LeapYear {
                public static void main(String[] args) {
                    Scanner scanner = new Scanner(System.in);
                    System.out.print("Enter a year: ");
                    int year = scanner.nextInt();
                    
                    if ((year % 4 == 0 && year % 100 != 0) || (year % 400 == 0)) {
                        System.out.println(year + " is a leap year.");
                    } else {
                        System.out.println(year + " is not a leap year.");
                    }
                }
            }

    Output: Enter a year: 2020, 2020 is a leap year.
