    • 18. Program to find all subsets of a string. 
                  import java.util.ArrayList;
            
            public class SubsetsOfString {
                public static void main(String[] args) {
                    String str = "abc";
                    ArrayList subsets = new ArrayList<>();
                    int n = str.length();
            
                    for (int i = 0; i < (1 << n); i++) {
                        StringBuilder subset = new StringBuilder();
                        for (int j = 0; j < n; j++) {
                            if ((i & (1 << j)) > 0) {
                                subset.append(str.charAt(j));
                            }
                        }
                        subsets.add(subset.toString());
                    }
            
                    System.out.println("Subsets: " + subsets);
                }
            }

      Output: Subsets: [, a, b, c, ab, ac, bc, abc]

    • 19. Program to find all the permutations of a string. 
                  import java.util.HashSet;
            
            public class Permutations {
                public static void main(String[] args) {
                    String str = "abc";
                    HashSet permutations = new HashSet<>();
                    permute("", str, permutations);
            
                    System.out.println("Permutations: " + permutations);
                }
            
                public static void permute(String prefix, String str, HashSet result) {
                    int n = str.length();
                    if (n == 0) {
                        result.add(prefix);
                    } else {
                        for (int i = 0; i < n; i++) {
                            permute(prefix + str.charAt(i), str.substring(0, i) + str.substring(i + 1, n), result);
                        }
                    }
                }
            }

      Output: Permutations: [abc, acb, bac, bca, cab, cba]

    • 20. Program to find maximum and minimum occurring character in a string. 
                  import java.util.HashMap;
            
            public class MaxMinChar {
                public static void main(String[] args) {
                    String str = "character";
                    HashMap charCount = new HashMap<>();
            
                    for (char ch : str.toCharArray()) {
                        charCount.put(ch, charCount.getOrDefault(ch, 0) + 1);
                    }
            
                    char maxChar = 0, minChar = 0;
                    int maxCount = Integer.MIN_VALUE, minCount = Integer.MAX_VALUE;
            
                    for (var entry : charCount.entrySet()) {
                        if (entry.getValue() > maxCount) {
                            maxCount = entry.getValue();
                            maxChar = entry.getKey();
                        }
                        if (entry.getValue() < minCount) {
                            minCount = entry.getValue();
                            minChar = entry.getKey();
                        }
                    }
            
                    System.out.println("Max Occurring Character: " + maxChar);
                    System.out.println("Min Occurring Character: " + minChar);
                }
            }

      Output: Max Occurring Character: c, Min Occurring Character: r
