    • 4. Java Program to Create 2 Radio Buttons and Display Selected Button Label 
                  import javax.swing.*;
            import java.awt.event.*;
            
            public class RadioButtonExample {
                public static void main(String[] args) {
                    JFrame frame = new JFrame("Radio Button Example");
                    JRadioButton rb1 = new JRadioButton("Option 1");
                    JRadioButton rb2 = new JRadioButton("Option 2");
                    rb1.setBounds(50, 50, 100, 30);
                    rb2.setBounds(50, 100, 100, 30);
                    ButtonGroup group = new ButtonGroup();
                    group.add(rb1);
                    group.add(rb2);
                    JLabel label = new JLabel();
                    label.setBounds(50, 150, 200, 30);
                    JButton button = new JButton("Show Selected");
                    button.setBounds(50, 200, 150, 30);
                    button.addActionListener(new ActionListener() {
                        public void actionPerformed(ActionEvent e) {
                            if (rb1.isSelected()) label.setText("Selected: Option 1");
                            else if (rb2.isSelected()) label.setText("Selected: Option 2");
                        }
                    });
                    frame.add(rb1);
                    frame.add(rb2);
                    frame.add(label);
                    frame.add(button);
                    frame.setSize(300, 300);
                    frame.setLayout(null);
                    frame.setVisible(true);
                }
            }

      Output: A window with two radio buttons and a button to display the selected option.

    • 18. Java Program to Display Several Dots on the Screen Continuously 
                  import javax.swing.*;
            import java.awt.*;
            
            public class DotsExample extends JPanel {
                public void paintComponent(Graphics g) {
                    super.paintComponent(g);
                    for (int i = 0; i < 10; i++) {
                        g.fillOval(i * 20, 50, 10, 10);
                    }
                }
            
                public static void main(String[] args) {
                    JFrame frame = new JFrame("Dots Example");
                    DotsExample dots = new DotsExample();
                    frame.add(dots);
                    frame.setSize(300, 200);
                    frame.setVisible(true);
                    while (true) {
                        dots.repaint();
                        try { Thread.sleep(500); } catch (InterruptedException e) {}
                    }
                }
            }

      Output: A window continuously displaying several dots.

    • 19. Java Program to Validate the TextField for only entering numbers using KeyListener 
                  import javax.swing.*;
            import java.awt.event.*;
            
            public class NumberValidationExample {
                public static void main(String[] args) {
                    JFrame frame = new JFrame("Number Validation Example");
                    JTextField textField = new JTextField();
                    textField.setBounds(50, 50, 150, 30);
                    textField.addKeyListener(new KeyAdapter() {
                        public void keyTyped(KeyEvent e) {
                            char c = e.getKeyChar();
                            if (!Character.isDigit(c)) {
                                e.consume();
                            }
                        }
                    });
                    frame.add(textField);
                    frame.setSize(300, 200);
                    frame.setLayout(null);
                    frame.setVisible(true);
                }
            }

      Output: A text field that only allows number input.

    • 20. Java Program to Validate the TextField for only entering email id using KeyAdapter 
                  import javax.swing.*;
            import java.awt.event.*;
            
            public class EmailValidationExample {
                public static void main(String[] args) {
                    JFrame frame = new JFrame("Email Validation Example");
                    JTextField textField = new JTextField();
                    textField.setBounds(50, 50, 200, 30);
                    textField.addKeyListener(new KeyAdapter() {
                        public void keyTyped(KeyEvent e) {
                            char c = e.getKeyChar();
                            if (!Character.isLetterOrDigit(c) && c != '@' && c != '.') {
                                e.consume();
                            }
                        }
                    });
                    frame.add(textField);
                    frame.setSize(300, 200);
                    frame.setLayout(null);
                    frame.setVisible(true);
                }
            }

      Output: A text field that only allows email characters.
