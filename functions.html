                  public class SmallestNumber {
                public static int findSmallest(int a, int b, int c) {
                    return Math.min(a, Math.min(b, c));
                }

    • 3. Write a Java method to display the middle character of a string. 
                  public class MiddleCharacter {
                public static String getMiddleCharacter(String str) {
                    int length = str.length();
                    if (length % 2 == 0) {
                        return String.valueOf(str.charAt(length / 2 - 1));
                    } else {
                        return str.charAt(length / 2 - 1) + "" + str.charAt(length / 2);
                    }
                }
            
                public static void main(String[] args) {
                    System.out.println(getMiddleCharacter("example")); // Output: am
                }
            }

      Output: am

    • 4. Write a Java method to count all the words in a string. 
                  public class WordCount {
                public static int countWords(String str) {
                    if (str == null || str.isEmpty()) {
                        return 0;
                    }
                    return str.split("\\s+").length;
                }
            
                public static void main(String[] args) {
                    System.out.println(countWords("Hello World! This is Java.")); // Output: 6
                }
            }

      Output: 6

    • 9. Write a Java method to find all twin prime numbers less than 100. 
                  public class TwinPrimes {
                public static void findTwinPrimes() {
                    for (int i = 3; i < 100; i++) {
                        if (isPrime(i) && isPrime(i + 2)) {
                            System.out.println("(" + i + ", " + (i + 2) + ")");
                        }
                    }
                }
            
                private static boolean isPrime(int num) {
                    for (int i = 2; i <= Math.sqrt(num); i++) {
                        if (num % i == 0) return false;
                    }
                    return true;
                }
            
                public static void main(String[] args) {
                    findTwinPrimes(); // Output: (3, 5), (5, 7), (11, 13), (17, 19), (29, 31), (41, 43), (59, 61), (71, 73), (79, 81)
                }
            }

      Output: (3, 5), (5, 7), (11, 13), (17, 19), (29, 31), (41, 43), (59, 61), (71, 73)
